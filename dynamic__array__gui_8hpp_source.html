165void GuiDynamicArray<T>::reserve(std::size_t capacity) {
166 capacity = std::min(capacity, static_cast<std::size_t>(8));
167 if (m_capacity > capacity) {
168 return;
169 }
170
171 auto* new_ptr = static_cast<GuiElement<T>*>(
172 ::operator new[](capacity * sizeof(GuiElement<T>)));
173 for (auto i = 0; i < m_size; ++i) {
174 ::new (&new_ptr[i]) GuiElement<T>(std::move(m_ptr[i]));
175 m_ptr[i].~GuiElement<T>();
176 }
177 for (auto i = m_size; i < capacity; ++i) {
178 ::new (&new_ptr[i]) GuiElement<T>();
179 new_ptr[i].set_index(i);
180 }
181
182 ::operator delete[](m_ptr);
183 m_ptr = new_ptr;
184 m_capacity = capacity;
185}
