107void GuiStack<T>::render_link(Vector2 src, Vector2 dest) {
108 constexpr int radius = GuiNode<T>::radius;
109 constexpr float scaled_len = radius / 8.0F;
110
111 // straight line
112 Vector2 link_pos{src.x - scaled_len, src.y + radius};
113 Vector2 link_size{2 * scaled_len, dest.y - src.y - 2 * radius};
114
115 // arrow
116 constexpr int arrow_size = scaled_len * 5;
117 Vector2 head{src.x, dest.y - radius + scaled_len / 2};
118 Vector2 side_left{head.x - arrow_size, head.y - arrow_size};
119 Vector2 side_right{head.x + arrow_size, head.y - arrow_size};
120
121 // draw both
122 DrawRectangleV(link_pos, link_size, Settings::get_instance().get_color(1));
123 DrawTriangle(head, side_right, side_left,
124 Settings::get_instance().get_color(1));
125}
