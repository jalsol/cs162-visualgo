107template<typename T>
108void GuiDoublyLinkedList<T>::render_link(Vector2 src, Vector2 dest) {
109 constexpr int radius = GuiNode<T>::radius;
110 constexpr float scaled_len = radius / 8.0F;
111
112 // straight line
113 Vector2 link_pos{src.x + radius, src.y - scaled_len};
114 Vector2 link_size{dest.x - src.x - 2 * radius, 2 * scaled_len};
115
116 // right arrow
117 constexpr int arrow_size = scaled_len * 5;
118 Vector2 right_head{dest.x - radius + scaled_len / 2, src.y};
119 Vector2 right_side_top{right_head.x - arrow_size,
120 right_head.y - arrow_size};
121 Vector2 right_side_bot{right_head.x - arrow_size,
122 right_head.y + arrow_size};
123
124 // left arrow
125 Vector2 left_head{src.x + radius - scaled_len / 2, src.y};
126 Vector2 left_side_top{left_head.x + arrow_size, left_head.y - arrow_size};
127 Vector2 left_side_bot{left_head.x + arrow_size, left_head.y + arrow_size};
128
129 // draw all
130 const Settings& settings = Settings::get_instance();
131 DrawRectangleV(link_pos, link_size, settings.get_color(1));
132 DrawTriangle(right_head, right_side_top, right_side_bot,
133 settings.get_color(1));
134 DrawTriangle(left_head, left_side_bot, left_side_top,
135 settings.get_color(1));
136}
