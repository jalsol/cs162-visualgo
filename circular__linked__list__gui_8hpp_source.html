115void GuiCircularLinkedList<T>::render_link(Vector2 src, Vector2 dest) {
116 constexpr int radius = GuiNode<T>::radius;
117 constexpr float scaled_len = radius / 8.0F;
118
119 // straight line
120 Vector2 link_pos{src.x + radius, src.y - scaled_len};
121 Vector2 link_size{dest.x - src.x - 2 * radius, 2 * scaled_len};
122
123 // arrow
124 constexpr int arrow_size = scaled_len * 5;
125 Vector2 head{dest.x - radius + scaled_len / 2, src.y};
126 Vector2 side_top{head.x - arrow_size, head.y - arrow_size};
127 Vector2 side_bot{head.x - arrow_size, head.y + arrow_size};
128
129 // draw both
130 const Settings& settings = Settings::get_instance();
131 DrawRectangleV(link_pos, link_size, settings.get_color(1));
132 DrawTriangle(head, side_top, side_bot, settings.get_color(1));

135template<typename T>
136void GuiCircularLinkedList<T>::render_back_link() {
137 if (m_head == nullptr && m_tail == nullptr) {

139 }
140
141 constexpr int num_points = 5;
142 const Vector2 head_pos = m_head->data.get_pos();
143 const Vector2 tail_pos = m_tail->data.get_pos();
144 constexpr int radius = GuiNode<T>::radius;
145 constexpr float scaled_len = radius / 8.0F;
146
147 std::array<Vector2, num_points> points{{
148 tail_pos,
149 {tail_pos.x + 2 * radius, tail_pos.y},
150 {tail_pos.x + 2 * radius, tail_pos.y + 3 * radius},
151 {head_pos.x, tail_pos.y + 3 * radius},
152 head_pos,
153 }};
154
155 constexpr int arrow_size = scaled_len * 5;
156 Vector2 head{head_pos.x, head_pos.y + radius - scaled_len / 2};
157 Vector2 side_left{head.x - arrow_size, head.y + arrow_size};
158 Vector2 side_right{head.x + arrow_size, head.y + arrow_size};
159
160 const Settings& settings = Settings::get_instance();
161 rlSetLineWidth(2 * scaled_len);
162 DrawLineStrip(points.data(), num_points, settings.get_color(1));
163 DrawTriangle(head, side_left, side_right, settings.get_color(1));
