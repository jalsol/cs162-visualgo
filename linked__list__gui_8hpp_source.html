107template<typename T>
108void GuiLinkedList<T>::render_link(Vector2 src, Vector2 dest) {
109 constexpr int radius = GuiNode<T>::radius;
110 constexpr float scaled_len = radius / 8.0F;
111
112 // straight line
113 Vector2 link_pos{src.x + radius, src.y - scaled_len};
114 Vector2 link_size{dest.x - src.x - 2 * radius, 2 * scaled_len};
115
116 // arrow
117 constexpr int arrow_size = scaled_len * 5;
118 Vector2 head{dest.x - radius + scaled_len / 2, src.y};
119 Vector2 side_top{head.x - arrow_size, head.y - arrow_size};
120 Vector2 side_bot{head.x - arrow_size, head.y + arrow_size};
121
122 // draw both
123 DrawRectangleV(link_pos, link_size, Settings::get_instance().get_color(1));
124 DrawTriangle(head, side_top, side_bot,
125 Settings::get_instance().get_color(1));
126}
