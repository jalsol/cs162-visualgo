136template<typename T>
137void GuiQueue<T>::render_link(Vector2 src, Vector2 dest) {
138 constexpr int radius = GuiNode<T>::radius;
139 constexpr float scaled_len = radius / 8.0F;
140
141 // straight line
142 Vector2 link_pos{src.x + radius, src.y - scaled_len};
143 Vector2 link_size{dest.x - src.x - 2 * radius, 2 * scaled_len};
144
145 // arrow
146 constexpr int arrow_size = scaled_len * 5;
147 Vector2 head{dest.x - radius + scaled_len / 2, src.y};
148 Vector2 side_top{head.x - arrow_size, head.y - arrow_size};
149 Vector2 side_bot{head.x - arrow_size, head.y + arrow_size};
150
151 // draw both
152 DrawRectangleV(link_pos, link_size, Settings::get_instance().get_color(1));
153 DrawTriangle(head, side_top, side_bot,
154 Settings::get_instance().get_color(1));
155}
